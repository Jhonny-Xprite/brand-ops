## Example Schema Structure

```typescript
// src/lib/schemas/files.ts
import { z } from 'zod'

export const UpdateFileMetadataSchema = z.object({
  type: z.string().min(1, 'File type is required'),
  status: z.string().min(1, 'Status is required'),
  tags: z.array(z.string()).max(20, 'Maximum 20 tags'),
  notes: z.string().max(500, 'Notes must be under 500 characters').optional(),
})

export type UpdateFileMetadata = z.infer<typeof UpdateFileMetadataSchema>
```

**API Usage:**
```typescript
// src/pages/api/files/[id].ts
const parsed = UpdateFileMetadataSchema.safeParse(req.body)
if (!parsed.success) {
  return res.status(400).json({ errors: parsed.error.errors })
}
const data = parsed.data
// ... proceed with validated data
```

**Form Usage:**
```typescript
import { zodResolver } from '@hookform/resolvers/zod'

const form = useForm({
  resolver: zodResolver(UpdateFileMetadataSchema),
  // ...
})
```
